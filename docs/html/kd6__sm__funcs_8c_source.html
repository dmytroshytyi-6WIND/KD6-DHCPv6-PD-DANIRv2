65  y = date.tm_year-2000;
66  //Month of year
67  m = date.tm_mon-1;
68  //Day of month
69  d = date.tm_mday-2;
70 
71  //January and February are counted as months 13 and 14 of the previous year
72  if(m <= 2)
73  {
74  m += 12;
75  y -= 1;
76  }
77  //Convert years to days
78  /* Sakamotos Algorithm.
79  * Explanation https://geeksforgeeks.org/tomohiko-sakamotos-algorithm-finging-day-week/
80  * Jan 1st 1 AD is a Monday in Gregorian calendar.
81  * Let us consider the first case in which we do not have leap years, hence
82  * total number of days in each year is 365.January has 31 days i.e 7*4+3 days
83  * so the day on 1st feb will always be 3 days ahead of the day on 1st January.
84  * Now february has 28 days(excluding leap years) which is exact multiple
85  * of 7 (7*4=28) Hence there will be no change in the month of march and it
86  * will also be 3 days ahead of the day on 1st January of that respective year.
87  * Considering this pattern, if we create an array of the leading number of days
88  * for each month then it will be given as t[] = {0, 3, 3, 6, 1, 4, 6, 2, 5, 0, 3, 5}.
89  * Now let us look at the real case when there are leap years. Every 4 years, our
90  * calculation will gain one extra day. Except every 100 years when it doesnt.
91  * Except every 400 years when it does. How do we put in these additional days?
92  * Well, just add y/4 <96> y/100 + y/400. Note that all division is integer division.
93  * This adds exactly the required number of leap days.But here is a problem, the leap
94  * day is 29 Feb and not 0 January.This means that the current year should not be
95  * counted for the leap day calculation for the first two months.Suppose that if
96  * the month were January or February, we subtracted 1 from the year. This means that
97  * during these months, the y/4 value would be that of the previous year and would
98  * not be counted. If we subtract 1 from the t[] values of every month after February?
99  * That would fill the gap, and the leap year problem is solved.That is, we need to make
100  * the following changes:
101  * 1.t[] now becomes {0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4}.
102  * 2.if m corresponds to Jan/Feb (that is, month<3) we decrement y by 1.
103  * 3.the annual increment inside the modulus is now y + y/4 <96> y/100 + y/400
104  * in place of y.
105  */
106  t = (365 * y) + (y / 4) - (y / 100) + (y / 400);
107  //Convert months to days
108  t += (30 * m) + (3 * (m + 1) / 5) + d;
109  //Convert days to seconds
110  t *= 86400;
111  //Add hours, minutes and seconds
112  t += (3600 * (date.tm_hour-1)) + (60 * date.tm_min) + date.tm_sec;
113 
114  //Return Unix time
115  return t;
116 }

610 
611  skb = skb_share_check(skb, GFP_ATOMIC);

620  //goto drop;
621  goto drop_unlock;
622  }

627  // goto drop;

634 
635  u8 kd6_msg_size =
636  skb->len -
637  (sizeof(struct ipv6hdr)+
638  sizeof(struct udphdr)+
639  4);//message type + transaction id

642  u8 *dhp;
643  dhp = (u8 *) udph + dh6_offset;
644  dhp+=1; //Transaction ID pointer
645 
646  u8 rx_xid[3];
647  memcpy(rx_xid,dhp,3*sizeof(u8));
648  if (memcmp(rx_xid, kd6_xid,3) != 0){
649  net_err_ratelimited("[KD6] Reply not for us, ,rx_xid[%x%x%x],internal_xid [%x%x%x]\n",

651  goto drop_unlock;
652  }
653 
654  u8 msg_type;
655  memcpy(&msg_type,dhp-=1,sizeof(u8));
656 
657  if (msg_type!=KD6_REPLY)
658  goto drop_unlock;
659  //pr_info ("[KD6] reply packed succesfully received");
660  kd6_strct->pkt_rcvd = true;

662  memcpy (&kd6_servaddr,&ipv6h->saddr,sizeof(struct in6_addr));
663  spin_unlock(&kd6_recv_lock);
664  return NF_DROP;
665 drop_unlock:
666  /* Show's over. Nothing to see here. */
667  spin_unlock(&kd6_recv_lock);
668 drop:
669  return NF_ACCEPT;
670 
671 }
